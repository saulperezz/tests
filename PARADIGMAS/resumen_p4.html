›
a) inserta-pos: insertar en posición específica
(define (inserta-pos dato pos lista) (if (= pos 0) (cons dato lista) (cons (first lista) (inserta-pos dato (- pos 1) (rest lista))))) (inserta-pos 'b 2 '(a a a a)) => '(a a b a a) (inserta-pos 'b 0 '(a a a a)) => '(b a a a a) (inserta-pos 'x 4 '(a b c d)) => '(a b c d x) (inserta-pos 1 0 '()) => '(1)

(define (coloca tres-listas un dos tres) (list (cons un (first tres-listas)) (cons dos (second tres-listas)) (cons tres (third tres-listas)))) (coloca '(() () ()) 'a 'b 'c) => '((a) (b) (c)) (coloca '((a) (b c) (d e f)) 'g 'h 'i) => '((g a) (h b c) (i d e f))
