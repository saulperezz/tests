(;; Encuentra hojas que cumplen predicado (define (cumplen-predicado pred estruct) (cond ((null? estruct) '()) ((number? (define (busca-mayores n estruct) (cumplen-predicado (lambda (x) (> x n)) estruct)) (busca-mayores 2 '(1 (2 3) ((4)))) => '(3 4)

(define (empieza-por car-buscado estruct) (cumplen-predicado (lambda (sym) (char=? (string-ref (symbol->string sym) 0) car-buscado)) estruct)) (empieza-por #\a '(a b (aardvark c) ((apple)))) => '(a aardvark apple)
