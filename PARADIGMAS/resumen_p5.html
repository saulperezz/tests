Ejercicio 1 — Composición y aplicación de funciones
›
a) aplica-veces: composición iterada
(;; Aplica n veces (f1 ∘ f2) sobre x (define (aplica-veces f1 f2 n x) (if (= n 0) x (aplica-veces f1 f2 (- n 1) (f1 (f2 x))))) (define doble (lambda (x) (* 2 x))) (define suma-2 (lambda (x) (+ x 2))) (aplica-veces (lambda (x) (+ x 1)) (lambda (x) (+ x 2)) 2 10) => 16 (aplica-veces doble suma-2 3 5) => 68
b) mueve-al-principio-condicion: búsqueda con predicado
(define (encuentra-primero pred lista) (cond ((null? lista) #f) ((pred (first lista)) (first lista)) (else (encuentra-primero pred (rest lista))))) (define (mueve-al-principio-condicion pred lista) (if (equal? (encuentra-primero pred lista) #f) lista (cons (encuentra-primero pred lista) (elimina-primera-cond pred lista)))) (mueve-al-principio-condicion number? '(a b c 1 d 1 e)) => '(1 a b c d 1 e) (mueve-al-principio-condicion string? '(a "hola" b "adios")) => '("hola" a b "adios")
c) comprueba: filtraje con predicado generalizado
(;; Filtra parejas (x y) donde (pred x y) sea #t (define (comprueba pred x-lista y-lista) (cond ((null? x-lista) '()) ((pred (first x-lista) (first y-lista)) (cons (cons (first x-lista) (first y-lista)) (comprueba pred (rest x-lista) (rest y-lista)))) (else (comprueba pred (rest x-lista) (rest y-lista))))) (comprueba (lambda (x y) (< x y)) '(1 2 3 4) '(2 1 4 3)) => '((1 . 2) (3 . 4))
Ejercicio 2 — Ordenación genérica
›
a) ordena-generica: ordenación con predicado personalizado
(define (inserta-generica pred elem lista-ord) (cond ((null? lista-ord) (list elem)) ((pred elem (first lista-ord)) (cons elem lista-ord)) (else (cons (first lista-ord) (inserta-generica pred elem (rest lista-ord)))))) (define (ordena-generica pred lista) (if (null? lista) '() (inserta-generica pred (first lista) (ordena-generica pred (rest lista))))) (ordena-generica < '(3 1 4 1 5)) => '(1 1 3 4 5)
b) ordena-por-longitud: ordenación de strings
(ordena-generica (lambda (s1 s2) (< (string-length s1) (string-length s2))) '("hi" "hello" "a")) => '("a" "hi" "hello")
Ejercicio 3 — Funciones de orden superior clásicas
›
a) map, filter, foldr, foldl: análisis
map: aplica función a cada elemento (map (lambda (x) (* x 2)) '(1 2 3)) => '(2 4 6) filter: selecciona elementos que cumplen predicado (filter (lambda (x) (> x 3)) '(1 5 2 8 3)) => '(5 8) foldr: reduce de derecha a izquierda (foldr + 0 '(1 2 3 4)) => 10 foldl: reduce de izquierda a derecha (foldl cons '() '(1 2 3)) => '(3 2 1)
b) apply: desempaquetado de lista
(apply + '(1 2 3 4)) => 10 (apply max '(3 1 4 1 5)) => 5
Ejercicio 4 — Reescritura con FOS
›
a) contar-datos-iguales-fos: con foldr
(define (contar-datos-iguales-fos dato lista) (foldr (lambda (x acc) (if (equal? x dato) (+ acc 1) acc)) 0 lista)) (contar-datos-iguales-fos 'a '(a b a c a)) => 3
b) expande-lista-fos: con foldr/append
(define (expande-lista-fos lista-parejas) (foldr (lambda (p acc) (append (expande-pareja p) acc)) '() lista-parejas))
c) comprueba-simbolos-fos: con filter/map
(define (comprueba-simbolos-fos lista-simbolos lista-num) (filter (lambda (par) (= (string-length (symbol->string (car par))) (cdr par))) (map (lambda (s n) (cons s n)) lista-simbolos lista-num)))
Ejercicio 5 — Búsqueda y cuantificación
›
a) busca-mayor: con comparador personalizado
(define (busca-mayor comparador lista) (if (null? (rest lista)) (first lista) (let ((mayor-resto (busca-mayor comparador (rest lista)))) (if (comparador (first lista) mayor-resto) (first lista) mayor-resto))))
b) todos-menores?: cuantificación universal
(define (todos-menores? n lista) (foldr (lambda (x acc) (and (< x n) acc)) #t lista)) (todos-menores? 5 '(1 2 3 4)) => #t (todos-menores? 3 '(1 2 3 4)) => #f
Ejercicio 6 — Truco de cartas mejorado
›
a) coloca mejorado: argumentos variadic
(define (coloca tres-listas . elementos) (map (lambda (lista elem) (cons elem lista)) tres-listas elementos))
b) escoge-en-orden: reorden de montones
(define (escoge-en-orden tres-listas orden-simbolo) (cond ((equal? orden-simbolo 'izquierda) (append (third tres-listas) (first tres-listas) (second tres-listas))) ((equal? orden-simbolo 'centro) (append (third tres-listas) (second tres-listas) (first tres-listas))) ((equal? orden-simbolo 'derecha) (append (second tres-listas) (third tres-listas) (first tres-listas)))))
c) junta-montones: con foldr/append
(define (junta-montones tres-listas) (foldr append '() tres-listas))
